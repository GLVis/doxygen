91 {
92  AUX_EVENTREC info;

213 static GLenum KeyDown(int key, GLenum status)
214 {
215  GLenum flag;
216  int i;
217 
218  flag = GL_FALSE;
219  if (keyTableCount) {
220  for (i = 0; i < keyTableCount; i++) {
221  if (key == keyTable[i].keyField) {
222  (*keyTable[i].KeyFunc)();
223  flag |= GL_TRUE;
224  }
225  }
226  }
227  if (modkeyTableCount) {
228  for (i = 0; i < modkeyTableCount; i++) {
229  if (key == modkeyTable[i].keyField) {
230  (*modkeyTable[i].KeyFunc)(status);
231  flag |= GL_TRUE;
232  }
233  }
234  }
235  return flag;
236 }
