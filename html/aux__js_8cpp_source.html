39 void Extrude1DMeshAndSolution(vis_data & data)
40 {
41  if (data.mesh->Dimension() != 1 || data.mesh->SpaceDimension() != 1)
42  {
43  return;
44  }
45 
46  // find xmin and xmax over the vertices of the 1D mesh
47  double xmin = numeric_limits<double>::infinity();
48  double xmax = -xmin;
49  for (int i = 0; i < data.mesh->GetNV(); i++)
50  {
51  const double x = data.mesh->GetVertex(i)[0];
52  if (x < xmin)
53  {
54  xmin = x;
55  }
56  if (x > xmax)
57  {
58  xmax = x;
59  }
60  }
61  Mesh *mesh2d = Extrude1D(data.mesh, 1, 0.1*(xmax - xmin));
62 
63  if (data.grid_f)
64  {
65  GridFunction *grid_f_2d =
66  Extrude1DGridFunction(data.mesh, mesh2d, data.grid_f, 1);
67  delete data.grid_f;
68  data.grid_f = grid_f_2d;
69  }
70  else if (data.sol.Size() == data.mesh->GetNV())
71  {
72  Vector sol2d(mesh2d->GetNV());
73  for (int i = 0; i < data.mesh->GetNV(); i++)
74  {
75  sol2d(2*i+0) = sol2d(2*i+1) = data.sol(i);
76  }
77  data.sol = sol2d;
78  }
79  data.mesh = mesh2d;
80 }

82 void setMeshSolution(vis_data & data)
83 {
84  if (1) // checkerboard solution
85  {
86  FiniteElementCollection *cfec;
87  if (data.mesh->Dimension() == 1)
88  {
89  cfec = new L2_FECollection(0, 1);
90  }
91  else if (data.mesh->Dimension() == 2)
92  {
93  cfec = new Const2DFECollection;
94  }
95  else
96  {
97  cfec = new Const3DFECollection;
98  }
99  FiniteElementSpace *cfes = new FiniteElementSpace(data.mesh, cfec);
100  data.grid_f = new GridFunction(cfes);
101  data.grid_f->MakeOwner(cfec);
102  {
103  Array<int> coloring;
104  srandom(time(0));
105  double a = double(random()) / (double(RAND_MAX) + 1.);
106  int el0 = (int)floor(a * data.mesh->GetNE());
107  cout << "Generating coloring starting with element " << el0+1
108  << " / " << data.mesh->GetNE() << endl;
109  data.mesh->GetElementColoring(coloring, el0);
110  for (int i = 0; i < coloring.Size(); i++)
111  {
112  (*data.grid_f)(i) = coloring[i];
113  }
114  cout << "Number of colors: " << data.grid_f->Max() + 1 << endl;
115  }
116  data.grid_f->GetNodalValues(data.sol);
117  // is_gf = 1;
118  }
119  else // zero solution
120  {
121  data.sol.SetSize (data.mesh->GetNV());
122  data.sol = 0.0;
123  }
124 }

208  data.mesh = new mfem::Mesh(mesh_file, 1, 0, fix_elem_orient);
209  data.grid_f = new mfem::GridFunction(data.mesh, mesh_file);
210  data.field_type = (data.grid_f->VectorDim() == 1) ? 0 : 1;
211  }
212  else if (data_type == "mesh")
213  {
214  data.mesh = new mfem::Mesh(mesh_file, 1, 0, fix_elem_orient);
215  setMeshSolution(data);
216  data.field_type = 2;
217  }
218  else if (data_type == "raw_scalar_2d")
219  {
220  Array<Array<double> *> vertices;
221  Array<Array<int> *> elements;
222  Array<int> elem_types;
223  string ident;
224  int num_patches, num_vert, num_elem, n;
225  mesh_file >> std::ws >> ident; // 'patches'
226  mesh_file >> num_patches;
227  // cout << ident << ' ' << num_patches << endl;
228  vertices.SetSize(num_patches);
229  vertices = NULL;
230  elements.SetSize(num_patches);
231  elements = NULL;
232  elem_types.SetSize(num_patches);
233  elem_types = 0;
234  int tot_num_vert = 0;
235  int tot_num_elem = 0;
236  int mesh_type = 0;
237  for (int i = 0; i < num_patches; i++)
238  {
239  mesh_file >> ws >> ident; // 'vertices'
240  mesh_file >> num_vert;
241  // cout << '\n' << ident << ' ' << num_vert << endl;
242  // read vertices in the format: x y z nx ny nz
243  vertices[i] = new Array<double>(6*num_vert);
244  Array<double> &verts = *vertices[i];
245  for (int j = 0; j < verts.Size(); j++)
246  {
247  mesh_file >> verts[j];
248  }
249 
250  mesh_file >> ws >> ident; // 'triangles' or 'quads'
251  if (ident == "triangles")
252  {
253  n = 3, mesh_type |= 1;
254  }
255  else
256  {
257  n = 4, mesh_type |= 2;
258  }
259  elem_types[i] = n;
260  mesh_file >> num_elem;
261  // cout << ident << ' ' << num_elem << endl;
262  elements[i] = new Array<int>(n*num_elem);
263  Array<int> &elems = *elements[i];
264  for (int j = 0; j < elems.Size(); j++)
265  {
266  mesh_file >> elems[j];
267  elems[j] += tot_num_vert;
268  }
269  tot_num_vert += num_vert;
270  tot_num_elem += num_elem;
271  }
272 
273  data.mesh = new mfem::Mesh(2, tot_num_vert, tot_num_elem, 0);
274  data.sol.SetSize(tot_num_vert);
275  data.normals.SetSize(3*tot_num_vert);
276 
277  int v_off = 0;
278  for (int i = 0; i < num_patches; i++)
279  {
280  Array<double> &verts = *vertices[i];
281  num_vert = verts.Size()/6;
282  for (int j = 0; j < num_vert; j++)
283  {
284  data.mesh->AddVertex(&verts[6*j]);
285  data.sol(v_off) = verts[6*j+2];
286  data.normals(3*v_off+0) = verts[6*j+3];
287  data.normals(3*v_off+1) = verts[6*j+4];
288  data.normals(3*v_off+2) = verts[6*j+5];
289  v_off++;
290  }
291 
292  n = elem_types[i];
293  Array<int> &elems = *elements[i];
294  num_elem = elems.Size()/n;
295  // int attr = 1;
296  int attr = i + 1;
297  if (n == 3)
298  for (int j = 0; j < num_elem; j++)
299  {
300  data.mesh->AddTriangle(&elems[3*j], attr);
301  }
302  else
303  for (int j = 0; j < num_elem; j++)
304  {
305  data.mesh->AddQuad(&elems[4*j], attr);
306  }
307  }
308 
309  if (mesh_type == 1)
310  {
311  data.mesh->FinalizeTriMesh(1, 0, fix_elem_orient);
312  }
313  else if (mesh_type == 2)
314  {
315  data.mesh->FinalizeQuadMesh(1, 0, fix_elem_orient);
316  }
317  else
318  {
319  mfem_error("Input data contains mixture of triangles and quads!");
320  }
321 
322  data.mesh->GenerateBoundaryElements();
323 
324  for (int i = num_patches; i > 0; )
325  {
326  i--;
327  delete elements[i];
328  delete vertices[i];
329  }
330 
331  data.field_type = 0;
332  }
333  else
334  {
335  data.field_type = -1;
336  cerr << "Unknown data format" << endl;
337  cerr << data_type << endl;
338  }
339 
340  if (data.field_type >= 0 && data.field_type <= 2)
341  {
342  Extrude1DMeshAndSolution(data);
343  }
344  return data;
345 }

347 // Replace a given VectorFiniteElement-based grid function (e.g. from a Nedelec
348 // or Raviart-Thomas space) with a discontinuous piece-wise polynomial Cartesian
349 // product vector grid function of the same order.
350 GridFunction *ProjectVectorFEGridFunction(GridFunction *gf)
351 {
352  if ((gf->VectorDim() == 3) && (gf->FESpace()->GetVDim() == 1))
353  {
354  int p = gf->FESpace()->GetOrder(0);
355  cout << "Switching to order " << p
356  << " discontinuous vector grid function..." << endl;
357  int dim = gf->FESpace()->GetMesh()->Dimension();
358  FiniteElementCollection *d_fec = new L2_FECollection(p, dim, 1);
359  FiniteElementSpace *d_fespace =
360  new FiniteElementSpace(gf->FESpace()->GetMesh(), d_fec, 3);
361  GridFunction *d_gf = new GridFunction(d_fespace);
362  d_gf->MakeOwner(d_fec);
363  gf->ProjectVectorFieldOn(*d_gf);
364  delete gf;
365  return d_gf;
366  }
367  return gf;
368 }

370 bool startVisualization(std::string input, std::string data_type, int w, int h)
371 {
372 
373  glvis_data = readStream(input, data_type);
374 
375  if (glvis_data.field_type < 0 || glvis_data.field_type > 2)
376  {
377  return false;
378  }
379 
380  if (InitVisualization("glvis", 0, 0, w, h))
381  {
382  return false;
383  }
384 
385  VisualizationSceneScalarData * vs;
386 
387  double mesh_range = -1.0;
388  if (glvis_data.field_type == 0 || glvis_data.field_type == 2)
389  {
390  if (glvis_data.grid_f)
391  {
392  glvis_data.grid_f->GetNodalValues(glvis_data.sol);
393  }
394  if (glvis_data.mesh->SpaceDimension() == 2)
395  {
396  VisualizationSceneSolution * vss;
397  if (glvis_data.field_type == 2)
398  {
399  paletteSet(4);
400  }
401  if (glvis_data.normals.Size() > 0)
402  {
403  vs = vss = new VisualizationSceneSolution(*glvis_data.mesh, glvis_data.sol,
404  &glvis_data.normals);
405  }
406  else
407  {
408  vs = vss = new VisualizationSceneSolution(*glvis_data.mesh, glvis_data.sol);
409  }
410  if (glvis_data.grid_f)
411  {
412  vss->SetGridFunction(*glvis_data.grid_f);
413  }
414  if (glvis_data.field_type == 2)
415  {
416  vs->OrthogonalProjection = 1;
417  vs->SetLight(0);
418  vs->Zoom(1.8);
419  }
420  }
421  else if (glvis_data.mesh->SpaceDimension() == 3)
422  {
423  VisualizationSceneSolution3d * vss;
424  vs = vss = new VisualizationSceneSolution3d(*glvis_data.mesh, glvis_data.sol);
425  if (glvis_data.grid_f)
426  {
427  vss->SetGridFunction(glvis_data.grid_f);
428  }
429  if (glvis_data.field_type == 2)
430  {
431  if (glvis_data.mesh->Dimension() == 3)
432  {
433  paletteSet(4);
434  // paletteSet(11);
435  // Set_Material_And_Light(4,3);
436  }
437  else
438  {
439  paletteSet(4);
440  }
441  vss->ToggleDrawAxes();
442  vss->ToggleDrawMesh();
443  }
444  }
445  if (glvis_data.field_type == 2)
446  {
447  if (glvis_data.grid_f)
448  {
449  mesh_range = glvis_data.grid_f->Max() + 1.0;
450  }
451  else
452  {
453  mesh_range = glvis_data.sol.Max() + 1.0;
454  }
455  }
456  }
457  else if (glvis_data.field_type == 1)
458  {
459  if (glvis_data.mesh->SpaceDimension() == 2)
460  {
461  if (glvis_data.grid_f)
462  {
463  vs = new VisualizationSceneVector(*glvis_data.grid_f);
464  }
465  else
466  {
467  vs = new VisualizationSceneVector(*glvis_data.mesh, glvis_data.solu,
468  glvis_data.solv);
469  }
470  }
471  else if (glvis_data.mesh->SpaceDimension() == 3)
472  {
473  if (glvis_data.grid_f)
474  {
475  glvis_data.grid_f = ProjectVectorFEGridFunction(glvis_data.grid_f);
476  vs = new VisualizationSceneVector3d(*glvis_data.grid_f);
477  }
478  else
479  {
480  vs = new VisualizationSceneVector3d(*glvis_data.mesh, glvis_data.solu,
481  glvis_data.solv, glvis_data.solw);
482  }
483  }
484  }
485 
486  if (vs)
487  {
488  // increase the refinement factors if visualizing a GridFunction
489  if (glvis_data.grid_f)
490  {
491  vs->AutoRefine();
492  vs->SetShading(2, true);
493  }
494  if (mesh_range > 0.0)
495  {
496  vs->SetValueRange(-mesh_range, mesh_range);
497  vs->SetAutoscale(0);
498  }
499  if (glvis_data.mesh->SpaceDimension() == 2 && glvis_data.field_type == 2)
500  {
501  SetVisualizationScene(vs, 2, glvis_data.keys.c_str());
502  }
503  else
504  {
505  SetVisualizationScene(vs, 3, glvis_data.keys.c_str());
506  }
507  }
508  SendExposeEvent();
509  return true;
510 }
