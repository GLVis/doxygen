214 void SendKeySequence (const char *seq)
215 {
216  const char *key = seq;
217 
218  for ( ; *key != '\0'; key++ ) // see /usr/include/X11/keysymdef.h
219  {
220  if ( ((*key - '0') < 10) && ((*key - '0') >= 0) ) // (keypad) number
221  {
222  SendKeyEvent(XK_0 + (*key) -'0');
223  continue;
224  }
225 
226  if ( ((*key - 'a') < 26) && ((*key - 'a') >= 0) ) // lowercase letter
227  {
228  SendKeyEvent(XK_a + (*key) -'a');
229  continue;
230  }
231 
232  if ( ((*key - 'A') < 26) && ((*key - 'A') >= 0) ) // uppercase letter
233  {
234  SendKeyEvent(XK_A + (*key) -'A',1);
235  continue;
236  }
237 
238  switch (*key)
239  {
240  case '+':
241  SendKeyEvent(XK_plus);
242  continue;
243  case '-':

553 inline void ComputeSphereAngles(int &newx, int &newy,
554  double &new_sph_u, double &new_sph_t)
555 {
556  GLint viewport[4];
557  double r, x, y, rr;
558  const double maxr = 0.996194698091745532295;
559 
560  glGetIntegerv(GL_VIEWPORT, viewport);
561  r = sqrt(sqr(viewport[2])+sqr(viewport[3]))*M_SQRT1_2;
562 
563  x = double(newx-viewport[0]-viewport[2]/2) / r;
564  y = double(-newy+viewport[1]+viewport[3]/2) / r;
565 
566  rr = sqrt(x*x+y*y);
567  if (rr > maxr)
568  {
569  x *= maxr/rr, y *= maxr/rr, rr = maxr;
570  }
571 
572  new_sph_u = 2.0 * acos(rr) - M_PI_2;
573  new_sph_t = atan2(y, x);
574 }

597 void LeftButtonLoc (AUX_EVENTREC *event)
598 {
599  GLint newx = event->data[AUX_MOUSEX];
600  GLint newy = event->data[AUX_MOUSEY];
601  int sendexpose = 1;
602 
603  if (event->data[2] & ControlMask)
604  {
605  if (event->data[2] & ShiftMask)
606  {
607  locscene->PreRotate(double(newx-oldx)/2, 0.0, 0.0, 1.0);
608  }
609  else
610  {
611  double new_sph_u, new_sph_t;
612 
613  ComputeSphereAngles(newx, newy, new_sph_u, new_sph_t);
614 
615  glMatrixMode (GL_MODELVIEW);
616  glLoadIdentity();
617 
618  double scoord[3], ncoord[3], inprod, cross[3];
619  scoord[0] = scoord[1] = cos(sph_u); scoord[2] = sin(sph_u);
620  ncoord[0] = ncoord[1] = cos(new_sph_u); ncoord[2] = sin(new_sph_u);
621  scoord[0] *= cos(sph_t); scoord[1] *= sin(sph_t);
622  ncoord[0] *= cos(new_sph_t); ncoord[1] *= sin(new_sph_t);
623  inprod = InnerProd(scoord, ncoord);
624  CrossProd(scoord, ncoord, cross);
625 
626  locscene->cam.GLMultTransposeRotMatrix();
627  glRotated(acos(inprod)*(180.0/M_PI), cross[0], cross[1], cross[2]);
628  glMultMatrixd(srot);
629  glGetDoublev(GL_MODELVIEW_MATRIX, locscene->rotmat);
630  }
631  }
632  else if (event->data[2] & Mod1Mask)
633  {
634  locscene->Rotate(double(newx-oldx)/2, 0.0, 0.0, 1.0);
635  }
636  else if (event->data[2] & ShiftMask)
637  {
638  locscene->Rotate(double(newx-oldx)/2, double(newy-oldy)/2);
639  }
640  else
641  {
642  locscene->Rotate(double(newy-oldy)/2, 1.0, 0.0, 0.0);
643  locscene->PreRotate(double(newx-oldx)/2, 0.0, 0.0, 1.0);
644  }
645 
646  oldx = newx;
647  oldy = newy;
648 
649  if (sendexpose)
650  {
651  SendExposeEvent();
652  }
653 }
