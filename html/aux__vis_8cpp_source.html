346 void MyReshape(GLsizei w, GLsizei h)
347 {
348  wnd->getRenderer().setViewport(w, h);
349 
350  gl3::GlMatrix projmtx;
351  projmtx.identity();
352 
353  double ViewCenterX = locscene->ViewCenterX;
354  double ViewCenterY = locscene->ViewCenterY;
355 
356  if (locscene->OrthogonalProjection)
357  {
358  double scale = locscene->ViewScale;
359  if (w <= h)
360  {
361  projmtx.ortho(-1.0, 1.0, -double(h)/w, double(h)/w, -10, 10);
362  }
363  else
364  {
365  projmtx.ortho(-double(w)/h, double(w)/h, -1, 1, -10, 10);
366  }
367  projmtx.scale(scale, scale, 1.0);
368  }
369  else
370  {
371  double ViewAngle = locscene->ViewAngle;
372 
373  if (w < h)
374  ViewAngle = (360.0/M_PI)*atan( tan( ViewAngle*(M_PI/360.0) ) *
375  double(h)/w );
376 
377  projmtx.perspective(ViewAngle, double(w)/h, 0.1, 5.0);
378  }
379  projmtx.translate(-ViewCenterX, -ViewCenterY, 0.0);
380  locscene->SetProjectionMtx(projmtx.mtx);
381 }

481 inline void ComputeSphereAngles(int &newx, int &newy,
482  double &new_sph_u, double &new_sph_t)
483 {
484  GLint viewport[4] = { 0, 0, 0, 0 };
485  double r, x, y, rr;
486  const double maxr = 0.996194698091745532295;
487 
488  wnd->getGLDrawSize(viewport[2], viewport[3]);
489  r = sqrt(sqr(viewport[2])+sqr(viewport[3]))*M_SQRT1_2;
490 
491  x = double(newx-viewport[0]-viewport[2]/2) / r;
492  y = double(-newy+viewport[1]+viewport[3]/2) / r;
493 
494  rr = sqrt(x*x+y*y);
495  if (rr > maxr)
496  {
497  x *= maxr/rr, y *= maxr/rr, rr = maxr;
498  }
499 
500  new_sph_u = 2.0 * acos(rr) - M_PI_2;
501  new_sph_t = atan2(y, x);
502 }

522 void LeftButtonLoc (EventInfo *event)
523 {
524  GLint newx = event->mouse_x;
525  GLint newy = event->mouse_y;
526  int sendexpose = 1;
527 
528  if (event->keymod & KMOD_CTRL)
529  {
530  if (event->keymod & KMOD_SHIFT)
531  {
532  locscene->PreRotate(double(newx-oldx)/2, 0.0, 0.0, 1.0);
533  }
534  else
535  {
536  double new_sph_u, new_sph_t;
537 
538  ComputeSphereAngles(newx, newy, new_sph_u, new_sph_t);
539 
540  gl3::GlMatrix newrot;
541  newrot.identity();
542 
543  double scoord[3], ncoord[3], inprod, cross[3];
544  scoord[0] = scoord[1] = cos(sph_u); scoord[2] = sin(sph_u);
545  ncoord[0] = ncoord[1] = cos(new_sph_u); ncoord[2] = sin(new_sph_u);
546  scoord[0] *= cos(sph_t); scoord[1] *= sin(sph_t);
547  ncoord[0] *= cos(new_sph_t); ncoord[1] *= sin(new_sph_t);
548  inprod = InnerProd(scoord, ncoord);
549  CrossProd(scoord, ncoord, cross);
550 
551  newrot.mult(locscene->cam.TransposeRotMatrix());
552  newrot.rotate(acos(inprod)*(180.0/M_PI), cross[0], cross[1], cross[2]);
553  newrot.mult(srot.mtx);
554  locscene->rotmat = newrot.mtx;
555  }
556  }
557  else if (event->keymod & KMOD_ALT)
558  {
559  locscene->Rotate(double(newx-oldx)/2, 0.0, 0.0, 1.0);
560  }
561  else if (event->keymod & KMOD_SHIFT)
562  {
563  locscene->Rotate(double(newx-oldx)/2, double(newy-oldy)/2);
564  }
565  else
566  {
567  locscene->Rotate(double(newy-oldy)/2, 1.0, 0.0, 0.0);
568  locscene->PreRotate(double(newx-oldx)/2, 0.0, 0.0, 1.0);
569  }
570 
571  oldx = newx;
572  oldy = newy;
573 
574  if (sendexpose)
575  {
576  SendExposeEvent();
577  }

765 int InvertSurfaceVertical(SDL_Surface *surface)
766 {
767  Uint8 *t, *a, *b, *last;
768  Uint16 pitch;
769 
770  if ( SDL_LOCKIFMUST(surface) < 0 )
771  {
772  return -2;
773  }
774 
775  /* do nothing unless at least two lines */
776  if (surface->h < 2)
777  {
778  SDL_UNLOCKIFMUST(surface);
779  return 0;
780  }
781 
782  /* get a place to store a line */
783  pitch = surface->pitch;
784  t = (Uint8*)malloc(pitch);
785 
786  if (t == NULL)
787  {
788  SDL_UNLOCKIFMUST(surface);
789  return -2;
790  }
791 
792  /* get first line; it's about to be trampled */
793  memcpy(t,surface->pixels,pitch);
794 
795  /* now, shuffle the rest so it's almost correct */
796  a = (Uint8*)surface->pixels;
797  last = a + pitch * (surface->h - 1);
798  b = last;
799 
800  while (a < b)
801  {
802  memcpy(a,b,pitch);
803  a += pitch;
804  memcpy(b,a,pitch);
805  b -= pitch;
806  }
807 
808  /* in this shuffled state, the bottom slice is too far down */
809  memmove( b, b+pitch, last-b );
810 
811  /* now we can put back that first row--in the last place */
812  memcpy(last,t,pitch);
813 
814  /* everything is in the right place; close up. */
815  free(t);
816  SDL_UNLOCKIFMUST(surface);
817 
818  return 0;
819 }
