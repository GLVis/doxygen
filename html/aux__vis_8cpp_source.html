00336 {
00337    glViewport (0, 0, w, h);
00338

00340    glLoadIdentity();

00343    double ViewCenterY = locscene->ViewCenterY;
00344 
00345    if (locscene->OrthogonalProjection)
00346    {

00349          glOrtho (-1.0, 1.0, -double(h)/w, double(h)/w, -10, 10);
00350       else
00351          glOrtho (-double(w)/h, double(w)/h, -1, 1, -10, 10);
00352       glScaled(scale, scale, 1.0);
00353       glTranslated(-ViewCenterX, -ViewCenterY, 0.0);
00354    }
00355    else
00356    {
00357       double ViewAngle = locscene->ViewAngle;
00358       double Distance  = 2.5;
00359 
00360       if (w < h)
00361          ViewAngle = (360.0/M_PI)*atan( tan( ViewAngle*(M_PI/360.0) ) *
00362                                         double(h)/w );
00363 
00364       gluPerspective(ViewAngle, double(w)/h, 0.4*Distance, 1.65*Distance);
00365       glTranslated(-ViewCenterX, -ViewCenterY, -Distance);

00465 inline void ComputeSphereAngles(int &newx, int &newy,
00466                                 double &new_sph_u, double &new_sph_t)
00467 {
00468    GLint viewport[4];
00469    double r, x, y, rr;
00470    const double maxr = 0.996194698091745532295;
00471 
00472    glGetIntegerv(GL_VIEWPORT, viewport);
00473    r = sqrt(sqr(viewport[2])+sqr(viewport[3]))*M_SQRT1_2;
00474 
00475    x = double(newx-viewport[0]-viewport[2]/2) / r;
00476    y = double(-newy+viewport[1]+viewport[3]/2) / r;
00477 
00478    rr = sqrt(x*x+y*y);
00479    if (rr > maxr)
00480       x *= maxr/rr, y *= maxr/rr, rr = maxr;
00481 
00482    new_sph_u = 2.0 * acos(rr) - M_PI_2;
00483    new_sph_t = atan2(y, x);
00484 }

00503 void LeftButtonLoc (AUX_EVENTREC *event)
00504 {
00505    GLint newx = event->data[AUX_MOUSEX];
00506    GLint newy = event->data[AUX_MOUSEY];
00507    int sendexpose = 1;
00508 
00509    if (event->data[2] & ControlMask)
00510    {
00511       if (event->data[2] & ShiftMask)
00512       {
00513          glMatrixMode (GL_MODELVIEW);
00514          glLoadMatrixd (locscene -> rotmat);
00515          glRotated ( double (newx-oldx) / 2, 0.0, 0.0, 1.0 );
00516          glGetDoublev (GL_MODELVIEW_MATRIX, locscene -> rotmat);
00517       }
00518       else
00519       {
00520          double new_sph_u, new_sph_t;
00521 
00522          ComputeSphereAngles(newx, newy, new_sph_u, new_sph_t);
00523 
00524          glMatrixMode (GL_MODELVIEW);
00525          glLoadIdentity();
00526 
00527          double scoord[3], ncoord[3], inprod, cross[3];
00528          scoord[0] = scoord[1] = cos(sph_u);     scoord[2] = sin(sph_u);
00529          ncoord[0] = ncoord[1] = cos(new_sph_u); ncoord[2] = sin(new_sph_u);
00530          scoord[0] *= cos(sph_t);     scoord[1] *= sin(sph_t);
00531          ncoord[0] *= cos(new_sph_t); ncoord[1] *= sin(new_sph_t);
00532          inprod = scoord[0]*ncoord[0]+scoord[1]*ncoord[1]+scoord[2]*ncoord[2];
00533          cross[0] =  scoord[1] * ncoord[2] - ncoord[1] * scoord[2];
00534          cross[1] =  ncoord[0] * scoord[2] - scoord[0] * ncoord[2];
00535          cross[2] =  scoord[0] * ncoord[1] - ncoord[0] * scoord[1];
00536          inprod = acos(inprod);
00537 
00538          glRotated(inprod*(180.0/M_PI), cross[0], cross[1], cross[2]);
00539          glMultMatrixd(srot);
00540          glGetDoublev (GL_MODELVIEW_MATRIX, locscene->rotmat);
00541       }
00542    }
00543    else if (event->data[2] & Mod1Mask)
00544    {
00545       glMatrixMode (GL_MODELVIEW);
00546       glLoadIdentity();
00547 
00548       glRotated(double (newx-oldx) / 2, 0.0, 0.0, 1.0);
00549       glMultMatrixd(locscene->rotmat);
00550       glGetDoublev (GL_MODELVIEW_MATRIX, locscene->rotmat);
00551    }
00552    else if (event->data[2] & ShiftMask)
00553    {
00554       locscene -> Rotate (double (newx-oldx) / 2, double (newy-oldy) / 2);
00555    }
00556    else
00557    {
00558       // locscene -> Rotate (double (newx-oldx) / 2, double (newy-oldy) / 2);
00559       glMatrixMode (GL_MODELVIEW);
00560       glLoadIdentity();
00561       glRotated ( double (newy-oldy) / 2, 1.0, 0.0, 0.0 );
00562       glMultMatrixd (locscene -> rotmat);
00563       glRotated ( double (newx-oldx) / 2, 0.0, 0.0, 1.0 );
00564       glGetDoublev (GL_MODELVIEW_MATRIX, locscene -> rotmat);
