375 void Set_AntiAliasing()
376 {
377  if (GetMultisample() > 0)
378  {
379  glEnable(GL_MULTISAMPLE);
380 
381 #ifdef GL_MULTISAMPLE_FILTER_HINT_NV
382  std::string s = (const char *)glGetString(GL_EXTENSIONS);
383  if (s.find("GL_NV_multisample_filter_hint") != std::string::npos)
384  {
385  glHint(GL_MULTISAMPLE_FILTER_HINT_NV, GL_NICEST);
386  }
387 #endif
388  }
389 
390  gl2psEnable(GL2PS_BLEND);
391  glEnable(GL_BLEND);
392  // glDisable(GL_DEPTH_TEST);
393  glBlendFunc(GL_SRC_ALPHA, GL_ONE_MINUS_SRC_ALPHA);
394  // glBlendFunc (GL_SRC_ALPHA_SATURATE, GL_ONE);
395 
396  glLineWidth(MS_LineWidth);
397 
398  // In order for polygon smoothing to work nicely, the polygons need to be
399  // sorted by depth. Therefore we leave polygon smoothing to multisampling.
400  if (0)
401  {
402  glEnable(GL_POLYGON_SMOOTH);
403  glHint(GL_POLYGON_SMOOTH_HINT, GL_NICEST);
404  }
405 
406  std::string vendor = (const char *)glGetString(GL_VENDOR);
407  if (vendor.find("ATI") == std::string::npos || GetMultisample() <= 0)
408  {
409  // In order for line smoothing to blend nicely with polygons, we draw
410  // the lines after the polygons.
411  glEnable(GL_LINE_SMOOTH);
412  glHint(GL_LINE_SMOOTH_HINT, GL_NICEST);
413  }
414  else
415  {
416 #ifdef GLVIS_DEBUG
417  std::cout <<
418  "Found 'ATI' in the GL vendor string:"
419  " using line smoothing via multisampling." << std::endl;
420 #endif
421  }
422 
423  glEnable(GL_POINT_SMOOTH);
424  glHint(GL_POINT_SMOOTH_HINT, GL_NICEST);
425 
426  /* add fog
427  glEnable(GL_AUTO_NORMAL);
428  glEnable(GL_NORMALIZE);
429  glEnable(GL_FOG);
430  GLfloat fogColor[4] = {1.0, 1.0, 1.0, 1.0};
431  GLint fogMode;
432  fogMode = GL_EXP2;
433  glFogi (GL_FOG_MODE, fogMode);
434  glFogfv (GL_FOG_COLOR, fogColor);
435  glFogf (GL_FOG_DENSITY, 2.0);
436  glHint (GL_FOG_HINT, GL_NICEST);
437  */
438 
439  AntiAliasing = 1;
440 }
