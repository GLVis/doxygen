57 void Camera::TurnUpDown(double angle)
58 {
59  double old_dir[3] = { dir[0], dir[1], dir[2] };
60  double c = cos(angle), s = sin(angle);
61  LinearCombination( c, old_dir, s, up, dir);
62  LinearCombination(-s, old_dir, c, up, up);
63  Normalize(dir);
64  ProjectVector(up, dir);
65 }
