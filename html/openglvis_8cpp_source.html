191 void VisualizationScene
192 ::DrawPatch(gl3::GlDrawable& drawable, const DenseMatrix &pts, Vector &vals,
193  DenseMatrix &normals,
194  const int n, const Array<int> &ind, const double minv,
195  const double maxv, const int normals_opt)
196 {
197  gl3::GlBuilder poly = drawable.createBuilder();
198  double na[3];
199 
200  if (normals_opt == 1 || normals_opt == -2)
201  {
202  normals.SetSize(3, pts.Width());
203  normals = 0.;
204  for (int i = 0; i < ind.Size(); i += n)
205  {
206  int j;
207  if (n == 3)
208  j = Compute3DUnitNormal(&pts(0, ind[i]), &pts(0, ind[i+1]),
209  &pts(0, ind[i+2]), na);
210  else
211  j = Compute3DUnitNormal(&pts(0, ind[i]), &pts(0, ind[i+1]),
212  &pts(0, ind[i+2]), &pts(0, ind[i+3]), na);
213  if (j == 0)
214  for ( ; j < n; j++)
215  for (int k = 0; k < 3; k++)
216  {
217  normals(k, ind[i+j]) += na[k];
218  }
219  }
220  }
221 
222  if (normals_opt != 0 && normals_opt != -1)
223  {
224  std::vector<gl3::VertexNormTex> vertices;
225  std::vector<int> indices;
226  vertices.reserve(pts.Size());
227  indices.reserve(ind.Size());
228  for (int i = 0; i < pts.Width(); i++)
229  {
230  vertices.emplace_back(
231  gl3::VertexNormTex
232  {
233  {(float) pts(0, i), (float) pts(1, i), (float) pts(2, i)},
234  {(float) normals(0, i), (float) normals(1, i), (float) normals(2, i)},
235  {(float) palette.GetColorCoord(vals(i), minv, maxv), 1.0 }
236  });
237  }
238  if (normals_opt > 0)
239  {
240  for (int i = 0; i < ind.Size(); i++)
241  {
242  indices.emplace_back(ind[i]);
243  }
244  }
245  else
246  {
247  for (int i = ind.Size()-1; i >= 0; i--)
248  {
249  indices.emplace_back(ind[i]);
250  }
251  }
252  if (n == 3)
253  {
254  drawable.addTriangleIndexed(vertices, indices);
255  }
256  else
257  {
258  drawable.addQuadIndexed(vertices, indices);
259  }
260  }
261  else
262  {
263  if (n == 3)
264  {
265  poly.glBegin(GL_TRIANGLES);
266  }
267  else
268  {
269  poly.glBegin(GL_QUADS);
270  }
271  for (int i = 0; i < ind.Size(); i += n)
272  {
273  int j;
274  if (n == 3)
275  j = Compute3DUnitNormal(&pts(0, ind[i]), &pts(0, ind[i+1]),
276  &pts(0, ind[i+2]), na);
277  else
278  j = Compute3DUnitNormal(&pts(0, ind[i]), &pts(0, ind[i+1]),
279  &pts(0, ind[i+2]), &pts(0, ind[i+3]), na);
280  if (j == 0)
281  {
282  if (normals_opt == 0)
283  {
284  poly.glNormal3dv(na);
285  for ( ; j < n; j++)
286  {
287  MySetColor(poly, vals(ind[i+j]), minv, maxv);
288  poly.glVertex3dv(&pts(0, ind[i+j]));
289  }
290  }
291  else
292  {
293  poly.glNormal3d(-na[0], -na[1], -na[2]);
294  for (j = n-1; j >= 0; j--)
295  {
296  MySetColor(poly, vals(ind[i+j]), minv, maxv);
297  poly.glVertex3dv(&pts(0, ind[i+j]));
298  }
299  }
300  }
301  }
302  poly.glEnd();
303  }
304 }
