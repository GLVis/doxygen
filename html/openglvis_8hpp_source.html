34 inline void CrossProd(const double a[], const double b[], double cp[])
35 {
36  cp[0] = a[1] * b[2] - a[2] * b[1];
37  cp[1] = a[2] * b[0] - a[0] * b[2];
38  cp[2] = a[0] * b[1] - a[1] * b[0];
39 }

41 inline int Normalize(double v[])
42 {
43  double len = sqrt(InnerProd(v, v));
44  if (len > 0.0)
45  {
46  len = 1.0 / len;
47  }
48  else
49  {
50  return 1;
51  }
52  for (int i = 0; i < 3; i++)
53  {
54  v[i] *= len;
55  }
56  return 0;
57 }

59 inline int ProjectVector(double v[], const double n[])
60 {
61  // project 'v' on the plane with normal given by 'n' and then normalize 'v'
62  LinearCombination(InnerProd(n, n), v, -InnerProd(v, n), n, v);
63  return Normalize(v);
64 }
