24 inline void LinearCombination(const double a, const double x[],
25  const double b, const double y[], double z[])
26 {
27  z[0] = a*x[0] + b*y[0];
28  z[1] = a*x[1] + b*y[1];
29  z[2] = a*x[2] + b*y[2];
30 }

32 inline double InnerProd(const double a[], const double b[])
33 {
34  return a[0] * b[0] + a[1] * b[1] + a[2] * b[2];
35 }
36 
37 inline void CrossProd(const double a[], const double b[], double cp[])
38 {
39  cp[0] = a[1] * b[2] - a[2] * b[1];
40  cp[1] = a[2] * b[0] - a[0] * b[2];
41  cp[2] = a[0] * b[1] - a[1] * b[0];
42 }

44 inline int Normalize(double v[])
45 {
46  double len = sqrt(InnerProd(v, v));
47  if (len > 0.0)
48  {
49  len = 1.0 / len;
50  }
51  else
52  {
53  return 1;
54  }
55  for (int i = 0; i < 3; i++)
56  {
57  v[i] *= len;
58  }
59  return 0;
60 }

62 inline int ProjectVector(double v[], const double n[])
63 {
64  // project 'v' on the plane with normal given by 'n' and then normalize 'v'
65  LinearCombination(InnerProd(n, n), v, -InnerProd(v, n), n, v);
66  return Normalize(v);
67 }
