76 void MeshRenderer::render(const RenderQueue& queue)
77 {
78  glClear(GL_COLOR_BUFFER_BIT | GL_DEPTH_BUFFER_BIT);
79  // elements containing opaque objects should be rendered first
80  RenderQueue sorted_queue = queue;
81  std::stable_partition(sorted_queue.begin(), sorted_queue.end(),
82  [](RenderQueue::value_type& renderPair)
83  {
84  return !renderPair.first.contains_translucent;
85  });
86  for (auto& q_elem : sorted_queue)
87  {
88  const RenderParams& params = q_elem.first;
89  device->setTransformMatrices(params.model_view.mtx, params.projection.mtx);
90  device->setMaterial(params.mesh_material);
91  device->setNumLights(params.num_pt_lights);
92  for (int i = 0; i < params.num_pt_lights; i++)
93  {
94  device->setPointLight(i, params.lights[i]);
95  }
96  device->setAmbientLight(params.light_amb_scene);
97  device->setStaticColor(params.static_color);
98  device->setClipPlaneUse(params.use_clip_plane);
99  device->setClipPlaneEqn(params.clip_plane_eqn);
100  // aggregate buffers with common parameters
101  std::vector<int> tex_bufs, no_tex_bufs;
102  std::vector<TextBuffer*> text_bufs;
103  GlDrawable* curr_drawable = q_elem.second;
104  for (int i = 0; i < NUM_LAYOUTS; i++)
105  {
106  for (size_t j = 0; j < GlDrawable::NUM_SHAPES; j++)
107  {
108  if (curr_drawable->buffers[i][j])
109  {
110  if (i == LAYOUT_VTX_TEXTURE0 || i == LAYOUT_VTX_NORMAL_TEXTURE0)
111  {
112  tex_bufs.emplace_back(curr_drawable->buffers[i][j].get()->getHandle());
113  }
114  else
115  {
116  no_tex_bufs.emplace_back(curr_drawable->buffers[i][j].get()->getHandle());
117  }
118  }
119  if (curr_drawable->indexed_buffers[i][j])
120  {
121  if (i == LAYOUT_VTX_TEXTURE0 || i == LAYOUT_VTX_NORMAL_TEXTURE0)
122  {
123  tex_bufs.emplace_back(curr_drawable->indexed_buffers[i][j].get()->getHandle());
124  }
125  else
126  {
127  no_tex_bufs.emplace_back(
128  curr_drawable->indexed_buffers[i][j].get()->getHandle());
129  }
130  }
131  }
132  }
133  text_bufs.emplace_back(&curr_drawable->text_buffer);
134  if (params.contains_translucent)
135  {
136  device->enableBlend();
137  }
138  else
139  {
140  device->enableDepthWrite();
141  }
142  device->attachTexture(GLDevice::SAMPLER_COLOR, color_tex);
143  device->attachTexture(GLDevice::SAMPLER_ALPHA, alpha_tex);
144  for (auto buf : tex_bufs)
145  {
146  device->drawDeviceBuffer(buf);
147  }
148  device->detachTexture(GLDevice::SAMPLER_COLOR);
149  device->detachTexture(GLDevice::SAMPLER_ALPHA);
150  for (auto buf : no_tex_bufs)
151  {
152  device->drawDeviceBuffer(buf);
153  }
154  if (!params.contains_translucent)
155  {
156  device->enableBlend();
157  device->disableDepthWrite();
158  }
159  device->attachTexture(1, font_tex);
160  device->setNumLights(0);
161  for (TextBuffer* buf : text_bufs)
162  {
163  device->drawDeviceBuffer(*buf);
164  }
165  device->enableDepthWrite();
166  if (!msaa_enable) { device->disableBlend(); }
167  }
168 }

270 void GLDevice::init()
271 {
272  // enable depth testing
273  glDepthFunc(GL_LEQUAL);
274  glEnable(GL_DEPTH_TEST);
275  // enable polygon offset to expose mesh lines
276  glPolygonOffset(1,1);
277  glEnable(GL_POLYGON_OFFSET_FILL);
278  // use "over" blending equation
279  glBlendFunc(GL_SRC_ALPHA, GL_ONE_MINUS_SRC_ALPHA);
280  glPixelStorei(GL_UNPACK_ALIGNMENT, 1);
281  // generate a white default texture modulation with default texture will just
282  // pass through input color
283  glBindTexture(GL_TEXTURE_2D, 0);
284  int black_color = 0xFFFFFFFF;
285  glTexImage2D(GL_TEXTURE_2D, 0, GL_RGBA, 1, 1, 0, GL_RGBA, GL_UNSIGNED_BYTE,
286  &black_color);
287  glTexParameteri(GL_TEXTURE_2D, GL_TEXTURE_WRAP_S, GL_CLAMP_TO_EDGE);
288  glTexParameteri(GL_TEXTURE_2D, GL_TEXTURE_WRAP_T, GL_CLAMP_TO_EDGE);
289  glTexParameteri(GL_TEXTURE_2D, GL_TEXTURE_MAG_FILTER, GL_NEAREST);
290  glTexParameteri(GL_TEXTURE_2D, GL_TEXTURE_MIN_FILTER, GL_NEAREST);
291 }

314 {
