96 std::string formatShaderString(const std::string &shader_string,
97  GLenum shader_type, int glsl_version)
98 {
99  std::string formatted = shader_string;
100 
101  // replace some identifiers depending on the version of glsl we're using
102  if (glsl_version >= 130)
103  {
104  if (shader_type == GL_VERTEX_SHADER)
105  {
106  formatted = std::regex_replace(formatted, std::regex("attribute"), "in");
107  formatted = std::regex_replace(formatted, std::regex("varying"), "out");
108  }
109 
110  else if (shader_type == GL_FRAGMENT_SHADER)
111  {
112  formatted = std::regex_replace(formatted, std::regex("varying"), "in");
113 
114  // requires GL_ARB_explicit_attrib_location extension or GLSL 3.30
115  // although gl_FragColor was deprecated in GLSL 1.3
116  if (glsl_version > 130 && glsl_version < 330)
117  {
118  formatted = "out vec4 fragColor;\n" + formatted;
119  formatted = std::regex_replace(formatted, std::regex("gl_FragColor"),
120  "fragColor");
121  }
122  else if (glsl_version >= 330)
123  {
124  formatted = "layout(location = 0) out vec4 fragColor;\n" + formatted;
125  formatted = std::regex_replace(formatted, std::regex("gl_FragColor"),
126  "fragColor");
127  }
128  }
129 
130  else
131  {
132  std::cerr << "buildShaderString: unknown shader type" << std::endl;
133  return {};
134  }
135 
136  formatted = std::regex_replace(formatted, std::regex("texture2D"), "texture");
137  }
138 
139  // add the header
140  formatted = std::regex_replace(GLSL_HEADER, std::regex("GLSL_VER"),
141  std::to_string(glsl_version)) + formatted;
142 
143  return formatted;
144 }

146 GLuint compileShaderString(const std::string &shader_str, GLenum shader_type)
147 {
148  int shader_len = shader_str.length();
149  const char *shader_cstr = shader_str.c_str();
150 
151  GLuint shader = glCreateShader(shader_type);
152  glShaderSource(shader, 1, &shader_cstr, &shader_len);
153  glCompileShader(shader);
154 
155  GLint stat;
156  glGetShaderiv(shader, GL_COMPILE_STATUS, &stat);
157  // glGetObjectParameteriv
158  if (stat == GL_FALSE)
159  {
160  std::cerr << "failed to compile shader" << std::endl;
161  int err_len;
162  glGetShaderiv(shader, GL_INFO_LOG_LENGTH, &err_len);
163  char *error_text = new char[err_len];
164  glGetShaderInfoLog(shader, err_len, &err_len, error_text);
165  std::cerr << error_text << std::endl;
166  delete[] error_text;
167  return 0;
168  }
169  return shader;
170 }

183 bool linkShaders(GLuint prgm, const std::vector<GLuint> &shaders)
184 {
185  // explicitly specify attrib positions so we don't have to reset VAO bindings
186  // when switching programs
187 
188  // for OSX, attrib 0 must be bound to render an object
189  glBindAttribLocation(prgm, CoreGLDevice::ATTR_VERTEX, "vertex");
190  glBindAttribLocation(prgm, CoreGLDevice::ATTR_TEXT_VERTEX, "textVertex");
191  glBindAttribLocation(prgm, CoreGLDevice::ATTR_NORMAL, "normal");
192  glBindAttribLocation(prgm, CoreGLDevice::ATTR_COLOR, "color");
193  glBindAttribLocation(prgm, CoreGLDevice::ATTR_TEXCOORD0, "texCoord0");
194 
195  for (GLuint i : shaders)
196  {
197  glAttachShader(prgm, i);
198  }
199  glLinkProgram(prgm);
200 
201  GLint stat;
202  glGetProgramiv(prgm, GL_LINK_STATUS, &stat);
203  if (stat == GL_FALSE)
204  {
205  cerr << "fatal: Shader linking failed" << endl;
206  }
207  return (stat == GL_TRUE);
208 }

210 bool CoreGLDevice::compileShaders()
211 {
212  std::string verStr = (char*)glGetString(GL_VERSION);
213  int ver_major, ver_minor;
214  int vs_idx = verStr.find_first_of(".");
215  ver_major = std::stoi(verStr.substr(vs_idx - 1, vs_idx));
216  ver_minor = std::stoi(verStr.substr(vs_idx + 1, 1));
217  int opengl_ver = ver_major * 100 + ver_minor * 10;
218  int glsl_ver = -1;
219 
220 #ifndef __EMSCRIPTEN__
221  // The GLSL version is the same as the OpenGL version when the OpenGL
222  // version is >= 3.30, otherwise it is:
223  //
224  // GL Version | GLSL Version
225  // -------------------------
226  // 2.0 | 1.10
227  // 2.1 | 1.20
228  // 3.0 | 1.30
229  // 3.1 | 1.40
230  // 3.2 | 1.50
231 
232  if (opengl_ver < 330)
233  {
234  if (ver_major == 2)
235  {
236  glsl_ver = opengl_ver - 90;
237  }
238  else if (ver_major == 3)
239  {
240  glsl_ver = opengl_ver - 170;
241  }
242  else
243  {
244  std::cerr << "fatal: unsupported OpenGL version " << opengl_ver << std::endl;
245  return false;
246  }
247  }
248  else
249  {
250  glsl_ver = opengl_ver;
251  }
252 #else
253  // GL Version | WebGL Version | GLSL Version
254  // 2.0 | 1.0 | 1.00 ES
255  // 3.0 | 2.0 | 3.00 ES
256  // 3.1 | | 3.10 ES
257  if (opengl_ver < 300)
258  {
259  glsl_ver = 100;
260  }
261  else
262  {
263  glsl_ver = 300;
264  }
265 #endif
266  std::cerr << "Using GLSL " << glsl_ver << std::endl;
267 
268  GLuint default_vs = compileRawShaderString(DEFAULT_VS, GL_VERTEX_SHADER,
269  glsl_ver);
270  GLuint default_fs = compileRawShaderString(DEFAULT_FS, GL_FRAGMENT_SHADER,
271  glsl_ver);
272 
273  default_prgm = glCreateProgram();
274  if (!linkShaders(default_prgm, {default_vs, default_fs}))
275  {
276  std::cerr << "Failed to link shaders for the default shader program" <<
277  std::endl;
278  glDeleteProgram(default_prgm);
279  default_prgm = 0;
280  return false;
281  }
282 
283 #ifndef __EMSCRIPTEN__
284  if (GLEW_EXT_transform_feedback || GLEW_VERSION_3_0)
285  {
286  feedback_prgm = glCreateProgram();
287  const char * xfrm_varyings[] =
288  {
289  "gl_Position",
290  "fColor",
291  "fClipCoord",
292  };
293  glTransformFeedbackVaryings(feedback_prgm, 3, xfrm_varyings,
294  GL_INTERLEAVED_ATTRIBS);
295 
296  GLuint printing_vs = compileRawShaderString(PRINTING_VS, GL_VERTEX_SHADER,
297  glsl_ver);
298  GLuint printing_fs = compileRawShaderString(PRINTING_FS, GL_FRAGMENT_SHADER,
299  glsl_ver);
300 
301  if (!linkShaders(feedback_prgm, {printing_vs, printing_fs}))
302  {
303  std::cerr << "failed to link shaders for the printing program" << std::endl;
304  glDeleteProgram(feedback_prgm);
305  feedback_prgm = 0;
306  return false;
307  }
308  }
309 #endif
310 
311  return true;
312 }

314 void CoreGLDevice::initializeShaderState(CoreGLDevice::RenderMode mode)
315 {
316  GLuint curr_prgm = 0;
317  if (mode == RenderMode::Default)
318  {
319  glUseProgram(default_prgm);
320  curr_prgm = default_prgm;
321  }
322  else if (mode == RenderMode::Feedback)
323  {
324  glUseProgram(feedback_prgm);
325  curr_prgm = feedback_prgm;
326  }
327  uniforms.clear();
328  // initialize uniform map
329  int num_unifs;
330  glGetProgramiv(curr_prgm, GL_ACTIVE_UNIFORMS, &num_unifs);
331  std::unordered_map<std::string, bool> uniform_query{};
332  for (const auto& uf : unif_list)
333  {
334  uniform_query.emplace(uf, false);
335  }
336  for (int i = 0; i < num_unifs; i++)
337  {
338  const size_t max_length = 100;
339  char unif_name[max_length];
340  GLsizei name_length;
341  GLint gl_size;
342  GLenum gl_type;
343  glGetActiveUniform(curr_prgm, i, max_length, &name_length, &gl_size, &gl_type,
344  unif_name);
345  auto it = uniform_query.find(std::string(unif_name, name_length));
346  if (it == uniform_query.end())
347  {
348 #ifdef GLVIS_DEBUG
349  std::cerr << "Warning: unexpected uniform \""
350  << std::string(unif_name, name_length)
351  << "\" found in shader." << std::endl;
352 #endif
353  }
354  else
355  {
356  it->second = true;
357  GLuint uf_idx = glGetUniformLocation(curr_prgm, it->first.c_str());
358  uniforms.emplace(it->first, uf_idx);
359  }
360  }
361  for (const auto& uf : uniform_query)
362  {
363  if (!uf.second)
364  {
365 #ifdef GLVIS_DEBUG
366  std::cerr << "Uniform \"" << uf.first
367  << "\" missing in shader, ignoring." << std::endl;
368 #endif
369  // set uniform index to -1 so glUniform ignores data
370  uniforms.emplace(uf.first, -1);
371  }
372  }
373  glUniform1i(uniforms["colorTex"], 0);
374  glUniform1i(uniforms["alphaTex"], 1);
375  use_clip_plane = false;
376 }

506 void CoreGLDevice::bufferToDevice(TextBuffer &t_buf)
507 {
508  std::vector<float> buf_data;
509  float tex_w = GetFont()->getAtlasWidth();
510  float tex_h = GetFont()->getAtlasHeight();
511  for (auto &e : t_buf)
512  {
513  float pen_x = e.ox, pen_y = e.oy;
514  char prev_c = '\0';
515  for (char c : e.text)
516  {
517  const GlVisFont::glyph &g = GetFont()->GetTexChar(c);
518  pen_x += GetFont()->GetKerning(prev_c, c);
519  // note: subtract 1 to account for the padding in the texture glyphs
520  float cur_x = pen_x + g.bear_x - 1;
521  float cur_y = -pen_y - g.bear_y - 1;
522  pen_x += g.adv_x;
523  pen_y += g.adv_y;
524  if (!g.w || !g.h)
525  {
526  continue;
527  }
528  float tris[] =
529  {
530  e.rx, e.ry, e.rz, cur_x, -cur_y, g.tex_x, 0, 0,
531  e.rx, e.ry, e.rz, cur_x + g.w, -cur_y, g.tex_x + g.w / tex_w, 0, 0,
532  e.rx, e.ry, e.rz, cur_x, -cur_y - g.h, g.tex_x, g.h / tex_h, 0,
533  e.rx, e.ry, e.rz, cur_x + g.w, -cur_y, g.tex_x + g.w / tex_w, 0, 0,
534  e.rx, e.ry, e.rz, cur_x, -cur_y - g.h, g.tex_x, g.h / tex_h, 0,
535  e.rx, e.ry, e.rz, cur_x + g.w, -cur_y - g.h, g.tex_x + g.w / tex_w, g.h / tex_h, 0
536  };
537  buf_data.insert(buf_data.end(), tris, tris + 8 * 6);
538  prev_c = c;
539  }
540  }
541  if (buf_data.size() == 0) { return; }
542  if (t_buf.getHandle() == 0)
543  {
544  GLuint handle;
545  glGenBuffers(1, &handle);
546  t_buf.setHandle(handle);
547  }
548  glBindBuffer(GL_ARRAY_BUFFER, t_buf.getHandle());
549  glBufferData(GL_ARRAY_BUFFER, sizeof(float) * buf_data.size(), buf_data.data(),
550  GL_STATIC_DRAW);
551 }

576 void CoreGLDevice::drawDeviceBuffer(int hnd)
577 {
578  if (hnd == 0) { return; }
579  if (vbos[hnd].count == 0) { return; }
580  glBindBuffer(GL_ARRAY_BUFFER, vbos[hnd].vert_buf);
581  bool indexed = false;
582  if (vbos[hnd].elem_buf != 0)
583  {
584  glBindBuffer(GL_ELEMENT_ARRAY_BUFFER, vbos[hnd].elem_buf);
585  indexed = true;
586  }
587  if (vbos[hnd].layout == Vertex::layout
588  || vbos[hnd].layout == VertexNorm::layout)
589  {
590  glVertexAttrib4fv(ATTR_COLOR, static_color.data());
591  }
592  GLenum shape = vbos[hnd].shape;
593  int count = vbos[hnd].count;
594  switch (vbos[hnd].layout)
595  {
596  case Vertex::layout:
597  drawDeviceBufferImpl<Vertex>(shape, count, indexed);
598  break;
599  case VertexColor::layout:
600  drawDeviceBufferImpl<VertexColor>(shape, count, indexed);
601  break;
602  case VertexTex::layout:
603  drawDeviceBufferImpl<VertexTex>(shape, count, indexed);
604  break;
605  case VertexNorm::layout:
606  drawDeviceBufferImpl<VertexNorm>(shape, count, indexed);
607  break;
608  case VertexNormColor::layout:
609  drawDeviceBufferImpl<VertexNormColor>(shape, count, indexed);
610  break;
611  case VertexNormTex::layout:
612  drawDeviceBufferImpl<VertexNormTex>(shape, count, indexed);
613  break;
614  default:
615  cerr << "WARNING: Unhandled vertex layout " << vbos[hnd].layout << endl;
616  }

618 void CoreGLDevice::drawDeviceBuffer(const TextBuffer& t_buf)
619 {
620  if (t_buf.getHandle() == 0) { return; }
621  if (t_buf.count() == 0) { return; }
622  glUniform1i(uniforms["containsText"], GL_TRUE);
623  glEnableVertexAttribArray(ATTR_VERTEX);
624  glEnableVertexAttribArray(ATTR_TEXT_VERTEX);
625  glEnableVertexAttribArray(ATTR_TEXCOORD0);
626  glBindBuffer(GL_ARRAY_BUFFER, t_buf.getHandle());
627 
628  glVertexAttrib4fv(ATTR_COLOR, static_color.data());

643 inline FeedbackVertex XFBPostTransform(CoreGLDevice::ShaderXfbVertex v,
644  float half_w, float half_h)
645 {
646  glm::vec3 coord = glm::make_vec3(v.pos);
647  glm::vec4 color = glm::make_vec4(v.color);
648  // clip coords -> ndc
649  coord /= v.pos[3];
650  // ndc -> device coords
651  coord.x = half_w * coord.x + half_w;
652  coord.y = half_h * coord.y + half_h;
653  return { coord, color };
654 }

656 void CoreGLDevice::processTriangleXfbBuffer(CaptureBuffer& cbuf,
657  const vector<ShaderXfbVertex>& verts)
658 {
659  float half_w = vp_width * 0.5f;
660  float half_h = vp_height * 0.5f;
661  if (!use_clip_plane)
662  {
663  // all triangles into capture buf
664  for (size_t i = 0; i < verts.size(); i++)
665  {
666  cbuf.triangles.emplace_back(XFBPostTransform(verts[i], half_w, half_h));
667  }
668  return;
669  }
670  // clipping needed
671  for (size_t t_i = 0; t_i < verts.size() / 3; t_i++)
672  {
673  if (verts[t_i*3].clipCoord >= 0.f
674  && verts[t_i*3+1].clipCoord >= 0.f
675  && verts[t_i*3+2].clipCoord >= 0.f)
676  {
677  // triangle fully in the unclipped region
678  for (int vert_i = 0; vert_i < 3; vert_i++)
679  {
680  cbuf.triangles.emplace_back(
681  XFBPostTransform(verts[t_i*3 + vert_i], half_w, half_h));
682  }
683  }
684  else if (verts[3*t_i].clipCoord < 0.f
685  && verts[3*t_i+1].clipCoord < 0.f
686  && verts[3*t_i+2].clipCoord < 0.f)
687  {
688  // triangle fully in clipped region
689  continue;
690  }
691  else
692  {
693  // clip through middle of triangle
694  for (int vert_i = 0; vert_i < 3; vert_i++)
695  {
696  int i_a = 3*t_i+vert_i;
697  int i_b = 3*t_i+((vert_i+1) % 3);
698  int i_c = 3*t_i+((vert_i+2) % 3);
699  // find two points on the same side of clip plane
700  if ((verts[i_a].clipCoord < 0.f) == (verts[i_b].clipCoord < 0.f))
701  {
702  // pts a, b are on same side of clip plane, c on other side
703  // perspective-correct interpolation factors, needed for colors
704  float c_w_a = verts[i_a].clipCoord / verts[i_a].pos[3];
705  float c_w_b = verts[i_b].clipCoord / verts[i_b].pos[3];
706  float c_w_c = verts[i_c].clipCoord / verts[i_c].pos[3];
707  // compute clip pts
708  glm::vec4 pos[2], color[2];
709  // a --- n_0 --- c
710  pos[0] = (glm::make_vec4(verts[i_a].pos) * verts[i_c].clipCoord
711  - glm::make_vec4(verts[i_c].pos) * verts[i_a].clipCoord);
712  color[0] = (glm::make_vec4(verts[i_a].color) * c_w_c
713  - glm::make_vec4(verts[i_c].color) * c_w_a)
714  / (c_w_c - c_w_a);
715  // b --- n_1 --- c
716  pos[1] = (glm::make_vec4(verts[i_b].pos) * verts[i_c].clipCoord
717  - glm::make_vec4(verts[i_c].pos) * verts[i_b].clipCoord);
718  color[1] = (glm::make_vec4(verts[i_b].color) * c_w_c
719  - glm::make_vec4(verts[i_c].color) * c_w_b)
720  / (c_w_c - c_w_b);
721  for (int i = 0; i < 2; i++)
722  {
723  // perform transform to device coords
724  pos[i] /= pos[i].w;
725  pos[i].x *= half_w; pos[i].x += half_w;
726  pos[i].y *= half_h; pos[i].y += half_h;
727  }
728 
729  if (verts[i_c].clipCoord < 0.f)
730  {
731  // pts a, b are in clip plane
732  // create quadrilateral a -- n_0 -- n_1 -- b
733  cbuf.triangles.emplace_back(XFBPostTransform(verts[i_a], half_w, half_h));
734  cbuf.triangles.emplace_back(pos[0], color[0]);
735  cbuf.triangles.emplace_back(pos[1], color[1]);
736  cbuf.triangles.emplace_back(XFBPostTransform(verts[i_a], half_w, half_h));
737  cbuf.triangles.emplace_back(pos[1], color[1]);
738  cbuf.triangles.emplace_back(XFBPostTransform(verts[i_b], half_w, half_h));
739  }
740  else
741  {
742  // pt c is in clip plane
743  // add triangle c -- n_0 -- n_1
744  cbuf.triangles.emplace_back(XFBPostTransform(verts[i_c], half_w, half_h));
745  cbuf.triangles.emplace_back(pos[0], color[0]);
746  cbuf.triangles.emplace_back(pos[1], color[1]);
747  }
748  break;
749  }
750  }
751  }
752  }
753 }

755 void CoreGLDevice::processLineXfbBuffer(CaptureBuffer& cbuf,
756  const vector<ShaderXfbVertex>& verts)
757 {
758  float half_w = vp_width * 0.5f;
759  float half_h = vp_height * 0.5f;
760  for (size_t i = 0; i < verts.size(); i += 2)
761  {
762  if (!use_clip_plane ||
763  (verts[i].clipCoord >= 0.f && verts[i+1].clipCoord >= 0.f))
764  {
765  cbuf.lines.emplace_back(XFBPostTransform(verts[i], half_w, half_h));
766  cbuf.lines.emplace_back(XFBPostTransform(verts[i+1], half_w, half_h));
767  }
768  else if (verts[i].clipCoord < 0.f && verts[i+1].clipCoord < 0.f)
769  {
770  // outside of clip plane;
771  continue;
772  }
773  else
774  {
775  int i_a, i_b;
776  if (verts[i].clipCoord < 0.f)
777  {
778  // vertex i lies in clipped region
779  i_a = i+1;
780  i_b = i;
781  }
782  else // verts[i+1].clipCoord < 0.f
783  {
784  // vertex i+1 lies in clipped region
785  i_a = i;
786  i_b = i+1;
787  }
788  // compute new vertex (CbVa - CaVb), where Vb lies in the clipped region
789  ShaderXfbVertex clip_vert;
790  // perspective-correct interpolation factors for color
791  float c_w_a = verts[i_a].clipCoord / verts[i_a].pos[3];
792  float c_w_b = verts[i_b].clipCoord / verts[i_b].pos[3];
793  for (int j = 0; j < 4; j++)
794  {
795  clip_vert.pos[j] = verts[i_a].pos[j] * verts[i_b].clipCoord
796  - verts[i_b].pos[j] * verts[i_a].clipCoord;
797  clip_vert.color[j] = (verts[i_a].color[j] * c_w_b
798  - verts[i_b].color[j] * c_w_a)
799  / (c_w_b - c_w_a);
800  }
801  cbuf.lines.emplace_back(XFBPostTransform(clip_vert, half_w, half_h));
802  cbuf.lines.emplace_back(XFBPostTransform(verts[i_a], half_w, half_h));
803  }
804  }
805 }

808 void CoreGLDevice::captureXfbBuffer(
809  CaptureBuffer& cbuf, int hnd)
810 {
811  if (hnd == 0) { return; }
812  if (vbos[hnd].count == 0) { return; }
813  // allocate feedback buffer
814  int buf_size = vbos[hnd].count * sizeof(ShaderXfbVertex);
815  glBufferData(GL_TRANSFORM_FEEDBACK_BUFFER,
816  buf_size, nullptr, GL_STATIC_READ);
817  glBindBufferBase(GL_TRANSFORM_FEEDBACK_BUFFER, 0, feedback_vbo);
818  // Draw objects in feedback-only mode
819  glBeginTransformFeedback(vbos[hnd].shape);
820  drawDeviceBuffer(hnd);
821  glEndTransformFeedback();
822  // Read back feedback buffer
823  vector<ShaderXfbVertex> xfbBuf(buf_size);
824  glGetBufferSubData(GL_TRANSFORM_FEEDBACK_BUFFER, 0, buf_size, xfbBuf.data());
825  if (vbos[hnd].shape == GL_TRIANGLES)
826  {

830  {

832  }
833  else
834  {
835  std::cerr << "Warning: GL_POINTS handling not implemented in transform "
836  << "feedback processing" << std::endl;
837  }
838 }
