73  void WaitEvent()
74  {
75  int nstr, nfd = 1;
76  struct pollfd pfd[2];
77 
78  pfd[0].fd = ConnectionNumber(disp);
79  pfd[0].events = POLLIN;
80  pfd[0].revents = 0;
81  if (glvis_command && visualize == 1)
82  {
83  pfd[1].fd = glvis_command->ReadFD();
84  pfd[1].events = POLLIN;
85  pfd[1].revents = 0;
86  nfd = 2;
87  }
88  do
89  {
90  nstr = poll(pfd, nfd, -1);
91  }
92  while (nstr == -1 && errno == EINTR);
93 
94  if (nstr == -1) { perror("poll()"); }
95  }
