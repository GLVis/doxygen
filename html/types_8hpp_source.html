231  void glEnd()
232  {
233  // create degenerate primitives if necessary
234  if (render_as == GL_LINES && count % 2 != 0)
235  {
236  saveVertex(curr);
237  }
238  if (render_as == GL_TRIANGLES)
239  {

241  {
242  saveVertex(curr);
243  }

254  void glVertex3d(double x, double y, double z)
255  {
256  curr.coords = { (float) x, (float) y, (float) z };
257  if (render_as == GL_LINES || render_as == GL_TRIANGLES)
258  {
259  // Lines and triangles are stored as-is
260  saveVertex(curr);
261  }
262  else if (is_line)
263  {
264  // LINE_LOOP and LINE_STRIP: each vertex creates a new line with the
265  // previous vertex
266  if (count == 0)
267  {
268  saved[0] = curr;
269  }
270  else
271  {
272  saveVertex(saved[1]);
273  saveVertex(curr);
274  }
275  saved[1] = curr;
276  }
277  else if (render_as == GL_QUADS)
278  {
279  if (count % 4 == 3)
280  {
281  // split on 0-2 diagonal
282  saveVertex(saved[0]);
283  saveVertex(saved[1]);
284  saveVertex(saved[2]);
285  saveVertex(saved[0]);
286  saveVertex(saved[2]);
287  saveVertex(curr);
288  count -= 4;
289  }
290  else
291  {
292  saved[count] = curr;
293  }
294  }
295  else if (render_as != GL_NONE)
296  {
297  // TriangleStrip, TriangleFan, Polygon
298  if (count >= 2)
299  {
300  saveVertex(saved[0]);
301  saveVertex(saved[1]);
302  saveVertex(curr);
303  if (render_as == GL_TRIANGLE_STRIP)
304  {
305  // pop earliest vertex
306  saved[0] = saved[1];
307  }
308  saved[1] = curr;
309  }
310  else
311  {
312  saved[count] = curr;
313  }
314  }
315  count++;
316  }

460  std::copy(verts.begin(), verts.end(), std::back_inserter(vertex_data));
461  int old_end = vertex_indices.size();

463  for (size_t i = old_end; i < vertex_indices.size(); i++)
464  {
465  vertex_indices[i] += index_offset;
