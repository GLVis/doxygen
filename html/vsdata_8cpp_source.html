00036    double xc = 0.5*(x[0]+x[1]);

00047    vy *= rlen*yscale;
00048    vz *= rlen*zscale;

01019 void VisualizationSceneScalarData::SetNewScalingFromBox()
01020 {
01021    // double eps = 1e-12;
01022    double eps = 0.0;
01023 
01024    // Find the new scaling
01025    if (scaling)
01026    {
01027       // Scale all sides of the box to 1.
01028       xscale = yscale = zscale = 1.;
01029       if ((x[1]-x[0])>eps) xscale /= (x[1]-x[0]);
01030       if ((y[1]-y[0])>eps) yscale /= (y[1]-y[0]);
01031       if ((z[1]-z[0])>eps) zscale /= (z[1]-z[0]);
01032    }
01033    else
01034    {
01035       // Find the largest side of the box in xscale
01036       xscale = x[1]-x[0];
01037       yscale = y[1]-y[0];
01038       zscale = z[1]-z[0];
01039       if (xscale < yscale) xscale = yscale;
01040       if (xscale < zscale) xscale = zscale;
01041       // Set proportional scaling so that the largest side of the box is 1.
01042       if (xscale > eps)
01043          xscale = ( 1.0 / xscale );
01044       else
01045          xscale = 1.0;
01046       zscale = yscale = xscale;
01047    }
01048 }
