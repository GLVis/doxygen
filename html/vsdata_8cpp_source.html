172  if (rhos == 0.0)

174  return;
175  }
176  double phi = acos(vz/rhos), theta = atan2(vy, vx);
177  const int n = 8;
178  const double step = 2*M_PI/n, nz = (1.0/4.0);
179  double point = step, cone[n+4][3], normal[n+2][3];
180  int i, j, k;
181 
182  cone[0][0] = 0; cone[0][1] = 0; cone[0][2] = 1;
183  cone[1][0] = cone_scale; cone[1][1] = 0; cone[1][2] = -4*cone_scale + 1;
184  normal[0][0] = 0.0/cone_scale;
185  normal[0][1] = 0.0/cone_scale;
186  normal[0][2] = 1.0/cone_scale;
187  normal[1][0] = 1.0/cone_scale;
188  normal[1][1] = 0.0/cone_scale;
189  normal[1][2] = nz/cone_scale;
190 
191  for (i=2; i<n+1; i++)
192  {
193  normal[i][0] = cos(point)/cone_scale;
194  normal[i][1] = sin(point)/cone_scale;
195  normal[i][2] = nz/cone_scale;
196 
197  cone[i][0] = cos(point)*cone_scale;
198  cone[i][1] = sin(point)*cone_scale;
199  cone[i][2] = -4*cone_scale + 1;
200  point += step;
201  }
202  cone[n+1][0] = cone_scale; cone[n+1][1] = 0; cone[n+1][2] =-4*cone_scale + 1;
203  normal[n+1][0] = 1.0/cone_scale;
204  normal[n+1][1] = 0.0/cone_scale;
205  normal[n+1][2] = nz/cone_scale;
206 
207  cone[n+2][0] = 0; cone[n+2][1] = 0; cone[n+2][2] = 0;
208  cone[n+3][0] = 0; cone[n+3][1] = 0; cone[n+3][2] = 1;
209 
210  if (arrow_scaling_type == 0)
211  {
212  length = rhos;
213  }
214 
215  // double xc = 0.5*(x[0]+x[1]), yc = 0.5*(y[0]+y[1]), zc = 0.5*(z[0]+z[1]);
216  double coord[3];
217  // double rlen = length/rhos;
218 
219  // px = (px-xc)*xscale; py = (py-yc)*yscale; pz = (pz-zc)*zscale;
220  // vx *= rlen*xscale; vy *= rlen*yscale; vz *= rlen*zscale;
221 
222  if (arrow_type == 1)
223  for (i=0; i<n+4; i++)
224  {
225  cone[i][2] -= 0.5;
226  }
227 
228  double M[3][3]= {{cos(theta)*cos(phi), -sin(theta), cos(theta)*sin(phi)},
229  {sin(theta)*cos(phi), cos(theta), sin(theta)*sin(phi)},
230  { -sin(phi), 0., cos(phi)}
231  };
232  double v[3] = { M[0][2]/xscale, M[1][2]/yscale, M[2][2]/zscale };
233  length /= sqrt(v[0]*v[0]+v[1]*v[1]+v[2]*v[2]);
234 
235  for (i=0; i<n+4; i++)
236  {
237  for (j=0; j<3; j++)
238  {
239  coord[j] = cone[i][j] * length;
240  }
241 
242  for (k=0; k<3; k++)
243  {
244  cone[i][k] = 0.;
245  for (j=0; j<3; j++)
246  {
247  cone[i][k] += M[k][j] * coord[j];
248  }
249  }
250  // cone[i][0] = (cone[i][0] + px)/xscale + xc;
251  // cone[i][1] = (cone[i][1] + py)/yscale + yc;
252  // cone[i][2] = (cone[i][2] + pz)/zscale + zc;
253  cone[i][0] = cone[i][0]/xscale + px;
254  cone[i][1] = cone[i][1]/yscale + py;
255  cone[i][2] = cone[i][2]/zscale + pz;
256  }
257 
258  for (i=0; i<=n+1; i++)
259  {
260  for (j=0; j<3; j++)
261  {
262  coord[j] = normal[i][j];
263  }
264 
265  for (k=0; k<3; k++)
266  {
267  normal[i][k] = 0.;
268  for (j=0; j<3; j++)
269  {

271  }
272  }

274  normal[i][1] *= yscale;
275  normal[i][2] *= zscale;
276  }
277 
278  glBegin(GL_TRIANGLE_FAN);
279  for (i=0; i<=n+1; i++)
280  {
281  glNormal3dv(normal[i]);
282  glVertex3dv(cone[i]);
283  }
284  glEnd();
285

287  glVertex3dv(cone[n+2]);
288  glVertex3dv(cone[n+3]);
289  glEnd();
