485 // 'v' must define three vectors that add up to the zero vector
486 // that is v[0], v[1], and v[2] are the sides of a triangle
487 int UnitCrossProd(double v[][3], double nor[])
488 {
489  // normalize the three vectors
490  for (int i = 0; i < 3; i++)
491  if (Normalize(v[i]))
492  {
493  return 1;
494  }
495 
496  // find the pair that forms an angle closest to pi/2, i.e. having
497  // the longest cross product:
498  double cp[3][3], max_a = 0.;
499  int k = 0;
500  for (int i = 0; i < 3; i++)
501  {
502  CrossProd(v[(i+1)%3], v[(i+2)%3], cp[i]);
503  double a = sqrt(InnerProd(cp[i], cp[i]));
504  if (max_a < a)
505  {
506  max_a = a, k = i;
507  }
508  }
509  if (max_a == 0.)
510  {
511  return 1;
512  }
513  for (int i = 0; i < 3; i++)
514  {
515  nor[i] = cp[k][i] / max_a;
516  }
517 
518  return 0;
519 }

521 int Compute3DUnitNormal(const double p1[], const double p2[],
522  const double p3[], double nor[])
523 {
524  double v[3][3];
525 
526  for (int i = 0; i < 3; i++)
527  {
528  v[0][i] = p2[i] - p1[i];
529  v[1][i] = p3[i] - p2[i];
530  v[2][i] = p1[i] - p3[i];
531  }
532 
533  return UnitCrossProd(v, nor);
534 }

536 int Compute3DUnitNormal (const double p1[], const double p2[],
537  const double p3[], const double p4[], double nor[])
538 {
539  double v[3][3];
540 
541  for (int i = 0; i < 3; i++)
542  {
543  // cross product of the two diagonals:
544  /*
545  v[0][i] = p3[i] - p1[i];
546  v[1][i] = p4[i] - p2[i];
547  v[2][i] = (p1[i] + p2[i]) - (p3[i] + p4[i]);
548  */
549 
550  // cross product of the two vectors connecting the midpoints of the
551  // two pairs of opposing sides; this gives the normal vector in the
552  // midpoint of the quad:
553  v[0][i] = 0.5 * ((p2[i] + p3[i]) - (p1[i] + p4[i]));
554  v[1][i] = 0.5 * ((p4[i] + p3[i]) - (p1[i] + p2[i]));
555  v[2][i] = p1[i] - p3[i];
556  }
557 
558  return UnitCrossProd(v, nor);
559 }
