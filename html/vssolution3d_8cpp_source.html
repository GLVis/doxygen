00421 // 'v' must define three vectors that add up to the zero vector
00422 // that is v[0], v[1], and v[2] are the sides of a triangle
00423 int UnitCrossProd(double v[][3], double nor[])
00424 {
00425    // normalize the three vectors
00426    for (int i = 0; i < 3; i++)
00427       if (Normalize(v[i]))
00428          return 1;
00429 
00430    // find the pair that forms an angle closest to pi/2, i.e. having
00431    // the longest cross product:
00432    double cp[3][3], max_a = 0.;
00433    int k = 0;
00434    for (int i = 0; i < 3; i++)
00435    {
00436       CrossProd(v[(i+1)%3], v[(i+2)%3], cp[i]);
00437       double a = sqrt(InnerProd(cp[i], cp[i]));
00438       if (max_a < a)
00439          max_a = a, k = i;
00440    }
00441    if (max_a == 0.)
00442       return 1;
00443    for (int i = 0; i < 3; i++)
00444       nor[i] = cp[k][i] / max_a;
00445 
00446    return 0;
00447 }

00464 int Compute3DUnitNormal (const double p1[], const double p2[],
00465                          const double p3[], const double p4[], double nor[])
00466 {
00467    double v[3][3];
00468 
00469    for (int i = 0; i < 3; i++)
00470    {
00471       // cross product of the two diagonals:
00472       /*
00473         v[0][i] = p3[i] - p1[i];
00474         v[1][i] = p4[i] - p2[i];
00475         v[2][i] = (p1[i] + p2[i]) - (p3[i] + p4[i]);
00476       */
00477 
00478       // cross product of the two vectors connecting the midpoints of the
00479       // two pairs of opposing sides; this gives the normal vector in the
00480       // midpoint of the quad:
00481       v[0][i] = 0.5 * ((p2[i] + p3[i]) - (p1[i] + p4[i]));
00482       v[1][i] = 0.5 * ((p4[i] + p3[i]) - (p1[i] + p2[i]));
00483       v[2][i] = p1[i] - p3[i];
00484    }
00485 
00486    return UnitCrossProd(v, nor);
00487 }

00835 void VisualizationSceneSolution3d::GetFaceNormals(
00836    const int FaceNo, const int side, const IntegrationRule &ir,
00837    DenseMatrix &normals)
00838 {
00839    // match the way GridFunction::GetFaceValues works
00840    double aJInv[9], alnor[3];
00841    DenseMatrix JInv(aJInv, 3, 3);
00842    Vector lnor(alnor, 3), nr;
00843    IntegrationRule eir(ir.GetNPoints());
00844    FaceElementTransformations *Tr;
00845    normals.SetSize(3, ir.GetNPoints());
00846    ElementTransformation *ETr;
00847    IntegrationPointTransformation *LTr;
00848    if (side == 0)
00849    {
00850       Tr = mesh->GetFaceElementTransformations(FaceNo, 5);
00851       ETr = Tr->Elem1;
00852       LTr = &Tr->Loc1;
00853    }
00854    else
00855    {
00856       Tr = mesh->GetFaceElementTransformations(FaceNo, 10);
00857       ETr = Tr->Elem2;
00858       LTr = &Tr->Loc2;
00859    }
00860    LTr->Transform(ir, eir);
00861    for (int i = 0; i < normals.Width(); i++)
00862    {
00863       LTr->Transf.SetIntPoint(&ir.IntPoint(i));
00864       const DenseMatrix &LJac = LTr->Transf.Jacobian();
00865       lnor(0) = LJac(1,0) * LJac(2,1) - LJac(2,0) * LJac(1,1);
00866       lnor(1) = LJac(2,0) * LJac(0,1) - LJac(0,0) * LJac(2,1);
00867       lnor(2) = LJac(0,0) * LJac(1,1) - LJac(1,0) * LJac(0,1);
00868       ETr->SetIntPoint(&eir.IntPoint(i));
00869       const DenseMatrix &Jac = ETr->Jacobian();
00870       CalcInverse(Jac, JInv);
00871       normals.GetColumnReference(i, nr);
00872       JInv.MultTranspose(lnor, nr);
00873    }
00874    if (side)
00875       normals *= -1.;
00876    JInv.ClearExternalData();
00877 }

00941    int n, DenseMatrix &pointmat, Vector &values, int *RG)
00942 {
00943    int i, j;
00944 
00945    if (FaceShiftScale == 0.0)
00946       return;
00947 
00948    double norm[3];
00949 
00950    if (RG == NULL) // use the normal vector of the cutting plane
00951    {
00952       double *eqn = CuttingPlane -> Equation();
00953       norm[0] = -eqn[0]; norm[1] = -eqn[1]; norm[2] = -eqn[2];
00954    }
00955    else
00956    {
00957       // use the normal vector to the polygon defined by
00958       // the points with indexes given by the first n integers in RG
00959       double pts[4][3];
00960       for (i = 0; i < n; i++)
00961          for (j = 0; j < 3; j++)
00962             pts[i][j] = pointmat(j,RG[i]);

00964          j = Compute3DUnitNormal (pts[0], pts[1], pts[2], pts[3], norm);

00969          // could not compute normal
00970          cerr << "WARNING: VisualizationSceneSolution3d::LiftRefinedSurf"
00971               << endl;
00972          return;
00973       }
00974    }
00975 
00976    double bbox_diam = sqrt ( (x[1]-x[0])*(x[1]-x[0]) +
00977                              (y[1]-y[0])*(y[1]-y[0]) +
00978                              (z[1]-z[0])*(z[1]-z[0]) );

00980 
00981    for (i = 0; i < pointmat.Width(); i++)
00982    {
00983       double val = sc * (values(i) - minv) / (maxv - minv);
00984       for (j = 0; j < 3; j++)
00985          pointmat(j, i) +=  val * norm[j];
00986    }

01524       { 0,1, 1,2, 3,2, 0,3, 4,5, 5,6, 7,6, 4,7, 0,4, 1,5, 2,6, 3,7 };

01526       {{ 3,  4,  6}, {17,  9, 18}, { 4,  6,  1}, {19, 11, 20},

01530        { 9, 18,  0}, { 7, 23, 14}, {11, 20,  2}, { 0, 17,  9},
01531        {12, 22,  5}, { 2, 19, 11}, {14, 16,  7}, { 5, 21, 12}};

01545          else

01548       {

01587             {
01588                n = n2;
01589             }
01590             else
01591             {
01592                break;
01593             }

01598          break;

01600          break;
01601       }
01602 
01603       while (n > 2)

01630                   pointmat(k,en[1]);
