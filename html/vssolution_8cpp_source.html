00204       }
00205       else
00206          update = 0;

00210    {

01499 void VisualizationSceneSolution::DrawCPLine(
01500    DenseMatrix &pointmat, Vector &values, Array<int> &ind)
01501 {
01502    int n, js, nv = ind.Size();
01503    double s, xs, ys;
01504 
01505    js = nv-1;
01506    xs = pointmat(0, ind[js]);
01507    ys = pointmat(1, ind[js]);
01508    s = CuttingPlane->Transform(xs, ys, 0.0);
01509    n = 0;
01510    for (int j = 0; j < nv; j++)
01511    {
01512       const double xt = pointmat(0, ind[j]);
01513       const double yt = pointmat(1, ind[j]);
01514       const double t = CuttingPlane->Transform(xt, yt, 0.0);
01515       if ((s >= 0.0 && t < 0.0) || (s < 0.0 && t >= 0.0))
01516       {
01517          double a = fabs(s) / (fabs(s) + fabs(t));
01518 
01519          glVertex3d((1.-a) * xs + a * xt,
01520                     (1.-a) * ys + a * yt,
01521                     (1.-a) * values(ind[js]) + a * values(ind[j]));
01522          n++;
01523       }
01524       s = t;
01525       js = j;
01526       xs = xt;
01527       ys = yt;
01528    }
01529    if (n != 2 && n != 4)
01530    {
01531       cerr << "n = " << n << endl;
01532       mfem_error("VisualizationSceneSolution::DrawCPLine");
01533    }
