00179 void KeyiPressed()
00180 {

00182    switch (refine_func)
00183    {
00184    case 0:
00185       vssol -> TimesToRefine += vssol -> EdgeRefineFactor;
00186       break;
00187    case 1:
00188       if (vssol -> TimesToRefine > vssol -> EdgeRefineFactor)
00189          vssol -> TimesToRefine -= vssol -> EdgeRefineFactor;
00190       else
00191          update = 0;
00192       break;
00193    case 2:
00194       vssol -> TimesToRefine /= vssol -> EdgeRefineFactor;
00195       vssol -> EdgeRefineFactor ++;
00196       vssol -> TimesToRefine *= vssol -> EdgeRefineFactor;
00197       break;
00198    case 3:
00199       if (vssol -> EdgeRefineFactor > 1)
00200       {
00201          vssol -> TimesToRefine /= vssol -> EdgeRefineFactor;
00202          vssol -> EdgeRefineFactor --;
00203          vssol -> TimesToRefine *= vssol -> EdgeRefineFactor;
00204       }
00205       else
00206          update = 0;
00207       break;
00208    }
00209    if (update && vssol -> shading == 2)
00210    {
00211       vssol -> FindNewBox();
00212       locscene -> PrepareAxes();
00213       vssol -> PrepareLines();
00214       vssol -> PrepareBoundary();
00215       locscene -> Prepare();
00216       vssol -> DefaultLevelLines();
00217       vssol -> PrepareLevelCurves();
00218       vssol -> PrepareCP();
00219       SendExposeEvent();

01544 void VisualizationSceneSolution::DrawCPLine(
01545    DenseMatrix &pointmat, Vector &values, Array<int> &ind)
01546 {
01547    int n, js, nv = ind.Size();
01548    double s, xs, ys;
01549 
01550    js = nv-1;
01551    xs = pointmat(0, ind[js]);
01552    ys = pointmat(1, ind[js]);
01553    s = CuttingPlane->Transform(xs, ys, 0.0);
01554    n = 0;
01555    for (int j = 0; j < nv; j++)
01556    {
01557       const double xt = pointmat(0, ind[j]);
01558       const double yt = pointmat(1, ind[j]);
01559       const double t = CuttingPlane->Transform(xt, yt, 0.0);
01560       if ((s >= 0.0 && t < 0.0) || (s < 0.0 && t >= 0.0))
01561       {
01562          double a = fabs(s) / (fabs(s) + fabs(t));
01563 
01564          glVertex3d((1.-a) * xs + a * xt,
01565                     (1.-a) * ys + a * yt,
01566                     (1.-a) * values(ind[js]) + a * values(ind[j]));
01567          n++;
01568       }
01569       s = t;
01570       js = j;
01571       xs = xt;
01572       ys = yt;
01573    }
01574    if (n != 2 && n != 4)
01575    {
