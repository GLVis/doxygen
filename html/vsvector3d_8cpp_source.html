261 void VisualizationSceneVector3d::SetScalarFunction()
262 {
263  FiniteElementSpace *fes = (VecGridF) ? VecGridF->FESpace() : NULL;
264 
265  switch (scal_func)
266  {
267  case 0: // magnitude
268  for (int i = 0; i < sol->Size(); i++)
269  (*sol)(i) = sqrt((*solx)(i) * (*solx)(i) +
270  (*soly)(i) * (*soly)(i) +
271  (*solz)(i) * (*solz)(i) );
272  if (GridF)
273  {
274  Array<int> dofs(3);
275  for (int i = 0; i < GridF->Size(); i++)
276  {
277  dofs.SetSize(1);
278  dofs[0] = i;
279  fes->DofsToVDofs(dofs);
280  double x = (*VecGridF)(dofs[0]);
281  double y = (*VecGridF)(dofs[1]);
282  double z = (*VecGridF)(dofs[2]);
283 
284  (*GridF)(i) = sqrt(x*x+y*y+z*z);
285  }
286  }
287  break;
288  case 1: // x-component
289  *sol = *solx;
290  if (GridF)
291  for (int i = 0; i < GridF->Size(); i++)
292  {
293  (*GridF)(i) = (*VecGridF)(fes->DofToVDof(i, 0));
294  }
295  break;
296  case 2: // y-component
297  *sol = *soly;
298  if (GridF)
299  for (int i = 0; i < GridF->Size(); i++)
300  {
301  (*GridF)(i) = (*VecGridF)(fes->DofToVDof(i, 1));
302  }
303  break;
304  case 3: // z-component
305  *sol = *solz;
306  if (GridF)
307  for (int i = 0; i < GridF->Size(); i++)
308  {
309  (*GridF)(i) = (*VecGridF)(fes->DofToVDof(i, 2));
310  }
311  break;
312  }
313  extra_caption = scal_func_name[scal_func];
314 }

565 void VisualizationSceneVector3d::PrepareFlat2()
566 {
567  int i, k, fn, fo, di, have_normals;
568  double bbox_diam, vmin, vmax;
569  int dim = mesh->Dimension();
570  int ne = (dim == 3) ? mesh->GetNBE() : mesh->GetNE();
571 
572  DenseMatrix pointmat, normals;
573  DenseMatrix vec_vals;
574  Vector values, normal;
575  RefinedGeometry * RefG;
576  Array<int> vertices;
577  double norm[3];
578  IsoparametricTransformation T;
579 
580  bbox_diam = sqrt( (x[1]-x[0])*(x[1]-x[0]) +
581  (y[1]-y[0])*(y[1]-y[0]) +
582  (z[1]-z[0])*(z[1]-z[0]) );
583  double sc = FaceShiftScale * bbox_diam;
584 
585  glNewList (displlist, GL_COMPILE);
586 
587  vmin = numeric_limits<double>::infinity();
588  vmax = -vmin;
589  for (i = 0; i < ne; i++)
590  {
591  if (dim == 3)
592  {
593  if (!bdr_attr_to_show[mesh->GetBdrAttribute(i)-1]) { continue; }
594 
595  if (cplane == 2)
596  {
597  // for cplane == 2, get vertices of the volume element, not bdr
598  int f, o, e1, e2;
599  mesh->GetBdrElementFace(i, &f, &o);
600  mesh->GetFaceElements(f, &e1, &e2);
601  mesh->GetElementVertices(e1, vertices);
602  }
603  else
604  {
605  mesh->GetBdrElementVertices(i, vertices);
606  }
607  }
608  else
609  {
610  if (!bdr_attr_to_show[mesh->GetAttribute(i)-1]) { continue; }
611 
612  mesh->GetElementVertices(i, vertices);
613  }
614 
615  if (cplane == 2 && CheckPositions(vertices)) { continue; }
616 
617  if (dim == 3)
618  {
619  mesh -> GetBdrElementFace (i, &fn, &fo);
620  RefG = GLVisGeometryRefiner.Refine(mesh -> GetFaceBaseGeometry (fn),
621  TimesToRefine);
622  // di = GridF->GetFaceValues(fn, 2, RefG->RefPts, values, pointmat);
623  di = fo % 2;
624  if (di == 1 && !mesh->FaceIsInterior(fn))
625  {
626  di = 0;
627  }
628  GridF->GetFaceValues(fn, di, RefG->RefPts, values, pointmat);
629  if (ianim > 0)
630  {
631  VecGridF->GetFaceVectorValues(fn, di, RefG->RefPts, vec_vals,
632  pointmat);
633  pointmat.Add(double(ianim)/ianimmax, vec_vals);
634  have_normals = 0;
635  }
636  else
637  {
638  GetFaceNormals(fn, di, RefG->RefPts, normals);
639  have_normals = 1;
640  }
641  ShrinkPoints(pointmat, i, fn, di);
642  }
643  else // dim == 2
644  {
645  RefG = GLVisGeometryRefiner.Refine(mesh->GetElementBaseGeometry(i),
646  TimesToRefine);
647  GridF->GetValues(i, RefG->RefPts, values, pointmat);
648  if (ianim > 0)
649  {
650  VecGridF->GetVectorValues(i, RefG->RefPts, vec_vals, pointmat);
651  pointmat.Add(double(ianim)/ianimmax, vec_vals);
652  have_normals = 0;
653  }
654  else
655  {
656  const IntegrationRule &ir = RefG->RefPts;
657  normals.SetSize(3, values.Size());
658  mesh->GetElementTransformation(i, &T);
659  for (int j = 0; j < values.Size(); j++)
660  {
661  T.SetIntPoint(&ir.IntPoint(j));
662  normals.GetColumnReference(j, normal);
663  CalcOrtho(T.Jacobian(), normal);
664  normal /= normal.Norml2();
665  }
666  have_normals = 1;
667  di = 0;
668  }
669  ShrinkPoints(pointmat, i, 0, 0);
670  }
671 
672  vmin = fmin(vmin, values.Min());
673  vmax = fmax(vmax, values.Max());
674 
675  int sides;
676  switch ((dim == 3) ? mesh->GetBdrElementType(i) : mesh->GetElementType(i))
677  {
678  case Element::TRIANGLE:
679  sides = 3;
680  break;
681 
682  case Element::QUADRILATERAL:
683  default:
684  sides = 4;
685  break;
686  }
687 
688  if (sc != 0.0 && have_normals)
689  {
690  for (int i = 0; i < 3; i++)
691  {
692  norm[i] = 0.0;
693  }
694  Normalize(normals);
695  for (k = 0; k < normals.Width(); k++)
696  for (int j = 0; j < 3; j++)
697  {
698  norm[j] += normals(j, k);
