00437    int i, const IntegrationRule &ir, Vector &vals, DenseMatrix &tr)
00438 {
00439    if (drawelems < 2)
00440    {
00441       DenseMatrix vec_vals;
00442       ElementTransformation *T;

00447       vals.SetSize(vec_vals.Width());
00448       for (int j = 0; j < vec_vals.Width(); j++)
00449       {
00450          if (Vec2Scalar == VecDivSubst || Vec2Scalar == VecCurlSubst ||
00451              Vec2Scalar == VecAnisotrSubst)
00452          {
00453             T = mesh->GetElementTransformation(i);
00454             T->SetIntPoint(&ir.IntPoint(j));
00455             if (Vec2Scalar == VecDivSubst)
00456             {
00457                vals(j) = VecGridF->GetDivergence(*T);
00458             }
00459             else
00460             {
00461                VecGridF->GetVectorGradient(*T, gv);
00462                if (Vec2Scalar == VecCurlSubst)
00463                {
00464                   vals(j) = gv(1, 0) - gv(0, 1); // curl v in 2D
00465                }
00466                else
00467                {
00468                   gv.Symmetrize();
00469                   gv.Eigenvalues(ev);
00470                   vals(j) = ev(1) - ev(0);
00471                   // vals(j) = ev(1);
00472                   // vals(j) = (ev(0) <= 0.0) ? ev(0) : ev(1);
00473                   // vals(j) = (fabs(ev(0)) >= fabs(ev(1))) ? ev(0) : ev(1);
00474                }
00475             }
00476          }
00477          else
00478             vals(j) = Vec2Scalar(vec_vals(0, j), vec_vals(1, j));
00479       }
00480    }
00481    else
00482    {
00483       ElementTransformation *T = mesh->GetElementTransformation(i);
00484 
00485       T->Transform(ir, tr);
00486 
00487       vals.SetSize(ir.GetNPoints());
00488       if (vc0.Size() == 0)
00489       {
00490          for (int j = 0; j < ir.GetNPoints(); j++)
00491          {
00492             T->SetIntPoint(&ir.IntPoint(j));
00493             vals(j) = T->Weight();
00494          }
00495       }
00496       else
00497       {
00498          mesh->GetElementTransformation(i, vc0, &T0);
00499          for (int j = 0; j < ir.GetNPoints(); j++)
00500          {
00501             T->SetIntPoint(&ir.IntPoint(j));
00502             T0.SetIntPoint(&ir.IntPoint(j));
00503             vals(j) = T->Weight() / T0.Weight();
00504          }
00505       }
00506 
00507       if (drawelems == 2)
00508       {
00509          for (int j = 0; j < vals.Size(); j++)
00510          {
00511             if (vals(j) <= 0.0)
00512             {
00513                vals = 0.0;
00514                break;
00515             }
00516             vals(j) = 1.0 / vals(j);
